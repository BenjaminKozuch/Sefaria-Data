13. Mishneh Torah, Torah Studya.1.1 ! א
נָשִׁים וַעֲבָדִים וּקְטַנִּים פְּטוּרִים מִתַּלְמוּד תּוֹרָה. <b>א</b> אֲבָל קָטָן אָבִיו חַיָּב [א] לְלַמְּדוֹ תּוֹרָה שֶׁנֶּאֱמַר  <small>דברים יא-יט</small> <small>וְלִמַּדְתֶּם אֹתָם אֶת בְּנֵיכֶם [ב] לְדַבֵּר בָּם</small>. וְאֵין הָאִשָּׁה חַיֶּבֶת לְלַמֵּד אֶת בְּנָהּ. שֶׁכָּל הַחַיָּב לִלְמֹד חַיָּב לְלַמֵּד: <small>(-)</small>
16. Mishneh Torah, Torah Studya.1.2 ! ב
<b>ב</b> כְּשֵׁם שֶׁחַיָּב אָדָם לְלַמֵּד אֶת בְּנוֹ כָּךְ הוּא חַיָּב לְלַמֵּד אֶת בֶּן בְּנוֹ שֶׁנֶּאֱמַר <small>דברים ד-ט</small> <small>וְהוֹדַעְתָּם לְבָנֶיךָ וְלִבְנֵי בָנֶיךָ</small>. וְלֹא בְּנוֹ וּבֶן בְּנוֹ בִּלְבַד אֶלָּא מִצְוָה עַל כָּל חָכָם וְחָכָם מִיִּשְׂרָאֵל לְלַמֵּד אֶת כָּל הַתַּלְמִידִים אַף עַל פִּי שֶׁאֵינָן בָּנָיו. שֶׁנֶּאֱמַר <small>דברים ו-ז</small> <small>וְשִׁנַּנְתָּם לְבָנֶיךָ</small> מִפִּי הַשְּׁמוּעָה לָמְדוּ בָּנֶיךָ אֵלּוּ תַּלְמִידֶיךָ שֶׁהַתַּלְמִידִים קְרוּיִין בָּנִים שֶׁנֶּאֱמַר <small>מלכים ב ב-ג</small> <small>וַיֵּצְאוּ בְנֵי הַנְּבִיאִים</small>. אִם כֵּן לָמָּה נִצְטַוָּה עַל בְּנוֹ וְעַל בֶּן בְּנוֹ. לְהַקְדִּים בְּנוֹ לְבֶן בְּנוֹ וּבֶן בְּנוֹ לְבֶן חֲבֵרוֹ:
19. Mishneh Torah, Torah Studya.1.3 ! ג
וְחַיָּב לִשְׂכֹּר מְלַמֵּד לִבְנוֹ לְלַמְּדוֹ. וְאֵינוֹ חַיָּב [ג] לְלַמֵּד בֶּן חֲבֵרוֹ אֶלָּא בְּחִנָּם. מִי שֶׁלֹּא לִמְּדוֹ אָבִיו חַיָּב לְלַמֵּד אֶת עַצְמוֹ כְּשֶׁיַּכִּיר שֶׁנֶּאֱמַר <small>דברים ה-א</small> <small>וּלְמַדְתֶּם אֹתָם וּשְׁמַרְתֶּם לַעֲשֹׂתָם</small>. וְכֵן אַתָּה מוֹצֵא בְּכָל מָקוֹם שֶׁהַתַּלְמוּד קוֹדֵם לַמַּעֲשֶׂה מִפְּנֵי שֶׁהַתַּלְמוּד מֵבִיא לִידֵי מַעֲשֶׂה וְאֵין הַמַּעֲשֶׂה מֵבִיא לִידֵי תַּלְמוּד:
22. Mishneh Torah, Torah Studya.1.4 ! ד
הָיָה הוּא רוֹצֶה לִלְמֹד תּוֹרָה וְיֵשׁ לוֹ בֵּן לִלְמֹד תּוֹרָה הוּא קוֹדֵם לִבְנוֹ. וְאִם הָיָה בְּנוֹ נָבוֹן וּמַשְׂכִּיל לְהָבִין מַה שֶּׁיִּלְמֹד יוֹתֵר מִמֶּנּוּ בְּנוֹ קוֹדֵם [ד]. וְאַף עַל פִּי שֶׁבְּנוֹ קוֹדֵם לֹא יִבָּטֵל הוּא. שֶׁכְּשֵׁם שֶׁמִּצְוָה עָלָיו לְלַמֵּד אֶת בְּנוֹ כָּךְ הוּא מְצֻוֶּה לְלַמֵּד עַצְמוֹ:
25. Mishneh Torah, Torah Studya.1.5 ! ה
לְעוֹלָם יִלְמֹד אָדָם תּוֹרָה וְאַחַר כָּךְ יִשָּׂא אִשָּׁה שֶׁאִם נָשָׂא אִשָּׁה תְּחִלָּה אֵין דַּעְתּוֹ פְּנוּיָה לִלְמֹד. וְאִם הָיָה יִצְרוֹ מִתְגַּבֵּר עָלָיו עַד שֶׁנִּמְצָא שֶׁאֵין לִבּוֹ פָּנוּי יִשָּׂא וְאַחַר כָּךְ יִלְמֹד תּוֹרָה:
28. Mishneh Torah, Torah Studya.1.6 ! ו
מֵאֵימָתַי אָבִיו חַיָּב לְלַמְּדוֹ תּוֹרָה. מִשֶּׁיַּתְחִיל לְדַבֵּר מְלַמְּדוֹ <small>דברים לג-ד</small> <small>תּוֹרָה צִוָּה לָנוּ משֶׁה</small> וּ<small>דברים ו-ד</small> <small>שְׁמַע יִשְׂרָאֵל</small>, וְאַחַר כָּךְ מְלַמְּדוֹ מְעַט מְעַט פְּסוּקִים פְּסוּקִים עַד שֶׁיִּהְיֶה בֶּן שֵׁשׁ אוֹ בֶּן שֶׁבַע הַכּל לְפִי בֻּרְיוֹ. וּמוֹלִיכוֹ אֵצֶל מְלַמֵּד הַתִּינוֹקוֹת:
31. Mishneh Torah, Torah Studya.1.7 ! ז
הָיָה מִנְהַג הַמְּדִינָה לִקַּח מְלַמֵּד הַתִּינוֹקוֹת שָׂכָר נוֹתֵן לוֹ שְׂכָרוֹ [ה]. וְחַיָּב לְלַמְּדוֹ בְּשָׂכָר עַד שֶׁיִּקְרָא תּוֹרָה שֶׁבִּכְתָב כֻּלָּהּ. מָקוֹם שֶׁנָּהֲגוּ לְלַמֵּד תּוֹרָה שֶׁבִּכְתָב בְּשָׂכָר מֻתָּר לְלַמֵּד בְּשָׂכָר. אֲבָל [ו] תּוֹרָה שֶׁבְּעַל פֶּה אָסוּר לְלַמְּדָהּ בְּשָׂכָר שֶׁנֶּאֱמַר <small>דברים ד-ה</small> <small>רְאֵה לִמַּדְתִּי אֶתְכֶם חֻקִּים וּמִשְׁפָּטִים כַּאֲשֶׁר צִוַּנִי ה'</small> וְגוֹ' מָה אֲנִי בְּחִנָּם לָמַדְתִּי אַף אַתֶּם לְמַדְתֶּם בְּחִנָּם מִמֶּנִּי וְכֵן כְּשֶׁתְּלַמְּדוּ לְדוֹרוֹת לַמְּדוּ בְּחִנָּם כְּמוֹ שֶׁלְּמַדְתֶּם מִמֶּנִּי. לֹא מָצָא מִי שֶׁיְּלַמְּדוֹ בְּחִנָּם יְלַמְּדוֹ בְּשָׂכָר שֶׁנֶּאֱמַר אֱמֶת קְנֵה. יָכוֹל יְלַמֵּד לַאֲחֵרִים בְּשָׂכָר. תַּלְמוּד לוֹמַר <small>משלי כג-כג</small> <small>וְאַל תִּמְכֹּר</small>. הָא לָמַדְתָּ שֶׁאָסוּר לוֹ לְלַמֵּד בְּשָׂכָר אַף עַל פִּי שֶׁלִּמְּדוֹ רַבּוֹ בְּשָׂכָר:
34. Mishneh Torah, Torah Studya.1.8 ! ח
כָּל אִישׁ מִיִּשְׂרָאֵל חַיָּב בְּתַלְמוּד תּוֹרָה בֵּין עָנִי בֵּין עָשִׁיר בֵּין שָׁלֵם בְּגוּפוֹ בֵּין בַּעַל יִסּוּרִין בֵּין בָּחוּר בֵּין שֶׁהָיָה זָקֵן גָּדוֹל שֶׁתָּשַׁשׁ כֹּחוֹ אֲפִלּוּ הָיָה עָנִי הַמִּתְפַּרְנֵס מִן הַצְּדָקָה וּמְחַזֵּר עַל הַפְּתָחִים וַאֲפִלּוּ בַּעַל אִשָּׁה וּבָנִים חַיָּב לִקְבֹּעַ לוֹ זְמַן לְתַלְמוּד תּוֹרָה בַּיּוֹם [ז] וּבַלַּיְלָה שֶׁנֶּאֱמַר <small>יהושע א-ח</small> <small>וְהָגִיתָ בּוֹ יוֹמָם וָלַיְלָה</small>:
37. Mishneh Torah, Torah Studya.1.9 ! ט
גְּדוֹלֵי חַכְמֵי יִשְׂרָאֵל הָיוּ מֵהֶן חוֹטְבֵי עֵצִים וּמֵהֶן שׁוֹאֲבֵי מַיִם וּמֵהֶן סוּמִים וְאַף עַל פִּי כֵן הָיוּ עוֹסְקִין בְּתַלְמוּד תּוֹרָה בַּיּוֹם וּבַלַּיְלָה וְהֵם מִכְּלַל מַעְתִּיקֵי הַשְּׁמוּעָה אִישׁ מִפִּי אִישׁ מִפִּי משֶׁה רַבֵּנוּ:
40. Mishneh Torah, Torah Studya.1.10 ! י
עַד אֵימָתַי חַיָּב לִלְמֹד תּוֹרָה עַד יוֹם מוֹתוֹ שֶׁנֶּאֱמַר <small>דברים ד-ט</small> <small>וּפֶן יָסוּרוּ מִלְּבָבְךָ כּל יְמֵי חַיֶּיךָ</small>. וְכָל זְמַן שֶׁלֹּא יַעֲסֹק בְּלִמּוּד הוּא שׁוֹכֵחַ:
43. Mishneh Torah, Torah Studya.1.11 ! יא
וְחַיָּב לְשַׁלֵּשׁ אֶת זְמַן לְמִידָתוֹ. שְׁלִישׁ בַּתּוֹרָה שֶׁבִּכְתָב. וּשְׁלִישׁ בַּתּוֹרָה שֶׁבְּעַל פֶּה. וּשְׁלִישׁ יָבִין וְיַשְׂכִּיל אַחֲרִית דָּבָר מֵרֵאשִׁיתוֹ וְיוֹצִיא דָּבָר מִדָּבָר וִידַמֶּה דָּבָר לְדָבָר וְיָבִין בַּמִּדּוֹת שֶׁהַתּוֹרָה נִדְרֶשֶׁת בָּהֶן עַד שֶׁיֵּדַע הֵיאַךְ הוּא עִקַּר הַמִּדּוֹת וְהֵיאַךְ יוֹצִיא הָאָסוּר וְהַמֻּתָּר וְכַיּוֹצֵא בָּהֶן מִדְּבָרִים שֶׁלָּמַד מִפִּי הַשְּׁמוּעָה. וְעִנְיָן זֶה הוּא הַנִּקְרָא גְּמָרָא:
46. Mishneh Torah, Torah Studya.1.12 ! יב
כֵּיצַד. הָיָה בַּעַל אֻמָּנוּת וְהָיָה עוֹסֵק בִּמְלַאכְתּוֹ שָׁלֹשׁ שָׁעוֹת בַּיּוֹם וּבַתּוֹרָה תֵּשַׁע. אוֹתָן הַתֵּשַׁע קוֹרֵא בְּשָׁלֹשׁ מֵהֶן בַּתּוֹרָה שֶׁבִּכְתָב וּבְשָׁלֹשׁ בַּתּוֹרָה שֶׁבְּעַל פֶּה [ח] וּבְשָׁלֹשׁ אֲחֵרוֹת מִתְבּוֹנֵן בְּדַעְתּוֹ לְהָבִין דָּבָר מִדָּבָר. וְדִבְרֵי קַבָּלָה בִּכְלַל תּוֹרָה שֶׁבִּכְתָב הֵן וּפֵרוּשָׁן בִּכְלַל תּוֹרָה שֶׁבְּעַל פֶּה. וְהָעִנְיָנִים הַנִּקְרָאִים פַּרְדֵּס בִּכְלַל הַגְּמָרָא הֵן. <b>א</b> בַּמֶּה דְּבָרִים אֲמוּרִים בִּתְחִלַּת תַּלְמוּדוֹ שֶׁל אָדָם אֲבָל כְּשֶׁיַּגְדִּיל בְּחָכְמָה וְלֹא יְהֵא צָרִיךְ לֹא לִלְמֹד תּוֹרָה שֶׁבִּכְתָב וְלֹא לַעֲסֹק תָּמִיד [ט] בַּתּוֹרָה שֶׁבְּעַל פֶּה יִקְרָא בְּעִתִּים מְזֻמָּנִים תּוֹרָה שֶׁבִּכְתָב וְדִבְרֵי הַשְּׁמוּעָה כְּדֵי שֶׁלֹּא יִשְׁכַּח דָּבָר מִדִּבְרֵי דִּינֵי תּוֹרָה וְיִפְנֶה כָּל יָמָיו לַגְּמָרָא בִּלְבַד לְפִי רֹחַב שֶׁיֵּשׁ בְּלִבּוֹ וְיִשּׁוּב דַּעְתּוֹ:
49. Mishneh Torah, Torah Studya.1.13 ! יג

מִי שֶׁנִּדּוּהוּ בְּפָנָיו אֵין מַתִּירִין לוֹ אֶלָּא בְּפָנָיו. נִדּוּהוּ שֶׁלֹּא בְּפָנָיו מַתִּירִין לוֹ בְּפָנָיו וְשֶׁלֹּא בְּפָנָיו. וְאֵין בֵּין נִדּוּי לַהֲפָרָה כְּלוּם אֶלָּא מְנַדִּין וּמַתִּירִין [ב] בְּרֶגַע אֶחָד כְּשֶׁיַּחֲזֹר הַמְנֻדֶּה לַמּוּטָב. וְאִם רָאוּ בֵּית דִּין לְהַנִּיחַ זֶה בְּנִדּוּי כַּמָּה שָׁנִים מַנִּיחִין כְּפִי רִשְׁעוֹ. וְכֵן אִם רָאוּ בֵּית דִּין לְהַחֲרִים לָזֶה לְכַתְּחִלָּה וּלְהַחֲרִים מִי שֶׁיֹּאכַל עִמּוֹ וְשׁוֹתֶה עִמּוֹ אוֹ מִי שֶׁיַּעֲמֹד עִמּוֹ בְּאַרְבַּע אַמּוֹת מַחְרִימִין כְּדֵי לְיַסְּרוֹ וּכְדֵי לַעֲשׂוֹת סְיָג לַתּוֹרָה עַד שֶׁלֹּא יִפְרְצוּ הַחוֹטְאִים. אַף עַל פִּי שֶׁיֵּשׁ רְשׁוּת לֶחָכָם לְנַדּוֹת לִכְבוֹדוֹ אֵינוֹ שֶׁבַח לְתַלְמִיד חָכָם לְהַנְהִיג עַצְמוֹ בְּדָבָר זֶה אֶלָּא מַעֲלִים אָזְנָיו מִדִּבְרֵי עַם הָאָרֶץ וְלֹא יָשִׁית לִבּוֹ לָהֶן כָּעִנְיָן שֶׁאָמַר שְׁלֹמֹה בְּחָכְמָתוֹ <small>קהלת ז-כא</small> <small>גַּם לְכָל הַדְּבָרִים אֲשֶׁר יְדַבְּרוּ אַל תִּתֵּן לִבְּךָ</small>. וְכֵן הָיָה דֶּרֶךְ חֲסִידִים הָרִאשׁוֹנִים שׁוֹמְעִים חֶרְפָּתָם וְאֵינָן מְשִׁיבִין וְלֹא עוֹד אֶלָּא שֶׁמּוֹחֲלִים לַמְחָרֵף וְסוֹלְחִים לוֹ. וַחֲכָמִים גְּדוֹלִים הָיוּ מִשְׁתַּבְּחִים בְּמַעֲשֵׂיהֶם הַנָּאִים וְאוֹמְרִים שֶׁמֵּעוֹלָם לֹא נִדּוּ אָדָם וְלֹא הֶחְרִימוּהוּ לִכְבוֹדָן. וְזוֹ הִיא דַּרְכָּם שֶׁל תַּלְמִידֵי חֲכָמִים שֶׁרָאוּי לֵילֵךְ בָּהּ. בַּמֶּה דְּבָרִים אֲמוּרִים כְּשֶׁבִּזָּהוּ אוֹ חֵרְפָהוּ בַּסֵּתֶר. אֲבָל תַּלְמִיד חָכָם שֶׁבִּזָּהוּ אוֹ חֵרְפוֹ אָדָם בְּפַרְהֶסְיָא אָסוּר לוֹ לִמְחל עַל כְּבוֹדוֹ וְאִם מָחַל נֶעֱנָשׁ שֶׁזֶּה בִּזְיוֹן תּוֹרָה אֶלָּא נוֹקֵם וְנוֹטֵר הַדָּבָר כְּנָחָשׁ עַד שֶׁיְּבַקֵּשׁ מִמֶּנּוּ מְחִילָה וְיִסְלַח לוֹ:
